A comprehensive database-driven logistics management system for tracking cameras, parts, manufacturers, and suppliers. Built with Java, JDBC, and SQL for enterprise-level supply chain operations.

System Overview

This logistics system provides complete visibility and management of camera equipment throughout the supply chain. It tracks inventory levels, manufacturer relationships, supplier management, and enables complex analytical queries for operational decision-making.

Core SQL Queries (10)

1. All Cameras in a Specific Store

SELECT c.CameraID, c.CameraName, c.Resolution, m.ManufacturerName
FROM CAMERA c
JOIN MANUFACTURER m ON c.ManufacturerID = m.ManufacturerID
WHERE c.StoreID = ?
ORDER BY c.CameraName;

Purpose: Retrieve all inventory at a specific store location

2. Count of Cameras by Manufacturer

SELECT m.ManufacturerName, COUNT(c.CameraID) as CameraCount
FROM MANUFACTURER m
LEFT JOIN CAMERA c ON m.ManufacturerID = c.ManufacturerID
GROUP BY m.ManufacturerID, m.ManufacturerName
ORDER BY CameraCount DESC;

Purpose: Analyze inventory distribution by manufacturer

3. Stores with Cameras from Specific Manufacturer

SELECT DISTINCT s.StoreID, s.StoreName, s.StoreCity, s.StoreState
FROM STORE s
JOIN CAMERA c ON s.StoreID = c.StoreID
JOIN MANUFACTURER m ON c.ManufacturerID = m.ManufacturerID
WHERE m.ManufacturerName = ?
ORDER BY s.StoreName;

Purpose: Find store locations carrying specific brands

4. Parts Suppliers by Manufacturer

SELECT m.ManufacturerName, ps.SupplierName, ps.PartType
FROM MANUFACTURER m
JOIN PARTS_SUPPLIER ps ON m.ManufacturerID = ps.ManufacturerID
ORDER BY m.ManufacturerName, ps.PartType;

Purpose: Map supply chain relationships

5. Total Inventory by Store Location

SELECT s.StoreName, s.StoreCity, s.StoreState, COUNT(c.CameraID) as InventoryCount
FROM STORE s
LEFT JOIN CAMERA c ON s.StoreID = c.StoreID
GROUP BY s.StoreID, s.StoreName, s.StoreCity, s.StoreState
ORDER BY InventoryCount DESC;

Purpose: Operational overview of inventory distribution

6. Cameras by Resolution Type

SELECT c.Resolution, COUNT(c.CameraID) as Count, 
       GROUP_CONCAT(c.CameraName) as Models
FROM CAMERA c
GROUP BY c.Resolution
ORDER BY Count DESC;

Purpose: Product mix analysis

7. Manufacturers with Supplies

SELECT DISTINCT m.ManufacturerID, m.ManufacturerName, m.ManufacturerCountry, 
       COUNT(ps.SupplierID) as SupplierCount
FROM MANUFACTURER m
LEFT JOIN PARTS_SUPPLIER ps ON m.ManufacturerID = ps.ManufacturerID
GROUP BY m.ManufacturerID, m.ManufacturerName, m.ManufacturerCountry;

Purpose: Supplier ecosystem analysis

8. Full Supply Chain Details for Camera

SELECT c.CameraName, c.Resolution, s.StoreName, s.StoreCity, 
       m.ManufacturerName, m.ManufacturerCountry, ps.SupplierName, ps.PartType
FROM CAMERA c
JOIN STORE s ON c.StoreID = s.StoreID
JOIN MANUFACTURER m ON c.ManufacturerID = m.ManufacturerID
LEFT JOIN PARTS_SUPPLIER ps ON m.ManufacturerID = ps.ManufacturerID
WHERE c.CameraID = ?;

Purpose: Complete supply chain traceability

9. Stores in Specific State with Camera Count

SELECT s.StoreName, s.StoreCity, s.StoreState, COUNT(c.CameraID) as CameraCount
FROM STORE s
LEFT JOIN CAMERA c ON s.StoreID = c.StoreID
WHERE s.StoreState = ?
GROUP BY s.StoreID, s.StoreName, s.StoreCity, s.StoreState
ORDER BY CameraCount DESC;

Purpose: Regional inventory analysis

10. Complex Join - Complete Operational View

SELECT s.StoreName, c.CameraName, c.Resolution, m.ManufacturerName, 
       m.ManufacturerCountry, ps.SupplierName, ps.PartType,
       COUNT(DISTINCT c.CameraID) as CameraCount
FROM STORE s
LEFT JOIN CAMERA c ON s.StoreID = c.StoreID
LEFT JOIN MANUFACTURER m ON c.ManufacturerID = m.ManufacturerID
LEFT JOIN PARTS_SUPPLIER ps ON m.ManufacturerID = ps.ManufacturerID
GROUP BY s.StoreID, c.CameraID, ps.SupplierID
ORDER BY s.StoreName, c.CameraName;

Purpose: Complete operational dashboard data

JDBC Implementation

Database Connection & Query Execution

import java.sql.*;
import javax.swing.*;
import javax.swing.table.DefaultTableModel;

public class LogisticsOperationsSystem {
    private static final String DB_URL = "jdbc:mysql://localhost:3306/logistics_db";
    private static final String USER = "root";
    private static final String PASSWORD = "password";

    // Initialize database connection
    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(DB_URL, USER, PASSWORD);
    }

    // Query: Get all cameras in a store
    public static void getCamerasInStore(int storeID, DefaultTableModel tableModel) {
        String query = "SELECT c.CameraID, c.CameraName, c.Resolution, m.ManufacturerName " +
                       "FROM CAMERA c " +
                       "JOIN MANUFACTURER m ON c.ManufacturerID = m.ManufacturerID " +
                       "WHERE c.StoreID = ? ORDER BY c.CameraName";
        
        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(query)) {
            
            pstmt.setInt(1, storeID);
            ResultSet rs = pstmt.executeQuery();
            
            tableModel.setRowCount(0);
            while (rs.next()) {
                tableModel.addRow(new Object[]{
                    rs.getInt("CameraID"),
                    rs.getString("CameraName"),
                    rs.getString("Resolution"),
                    rs.getString("ManufacturerName")
                });
            }
        } catch (SQLException e) {
            e.printStackTrace();
            JOptionPane.showMessageDialog(null, "Database error: " + e.getMessage());
        }
    }

    // Query: Count cameras by manufacturer
    public static void getCameraCountByManufacturer(DefaultTableModel tableModel) {
        String query = "SELECT m.ManufacturerName, COUNT(c.CameraID) as CameraCount " +
                       "FROM MANUFACTURER m " +
                       "LEFT JOIN CAMERA c ON m.ManufacturerID = c.ManufacturerID " +
                       "GROUP BY m.ManufacturerID, m.ManufacturerName " +
                       "ORDER BY CameraCount DESC";
        
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(query)) {
            
            tableModel.setRowCount(0);
            while (rs.next()) {
                tableModel.addRow(new Object[]{
                    rs.getString("ManufacturerName"),
                    rs.getInt("CameraCount")
                });
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    // Query: Inventory by store location
    public static void getInventoryByStore(DefaultTableModel tableModel) {
        String query = "SELECT s.StoreName, s.StoreCity, s.StoreState, " +
                       "COUNT(c.CameraID) as InventoryCount " +
                       "FROM STORE s " +
                       "LEFT JOIN CAMERA c ON s.StoreID = c.StoreID " +
                       "GROUP BY s.StoreID, s.StoreName, s.StoreCity, s.StoreState " +
                       "ORDER BY InventoryCount DESC";
        
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(query)) {
            
            tableModel.setRowCount(0);
            while (rs.next()) {
                tableModel.addRow(new Object[]{
                    rs.getString("StoreName"),
                    rs.getString("StoreCity"),
                    rs.getString("StoreState"),
                    rs.getInt("InventoryCount")
                });
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    // Add new camera to inventory
    public static void addCamera(String cameraName, String resolution, 
                                 int storeID, int manufacturerID) {
        String query = "INSERT INTO CAMERA (CameraName, Resolution, StoreID, ManufacturerID) " +
                       "VALUES (?, ?, ?, ?)";
        
        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(query)) {
            
            pstmt.setString(1, cameraName);
            pstmt.setString(2, resolution);
            pstmt.setInt(3, storeID);
            pstmt.setInt(4, manufacturerID);
            
            int rowsInserted = pstmt.executeUpdate();
            if (rowsInserted > 0) {
                JOptionPane.showMessageDialog(null, "Camera added successfully!");
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(null, "Error adding camera: " + e.getMessage());
        }
    }

    // Search cameras by manufacturer
    public static void searchByManufacturer(String manufacturerName, DefaultTableModel tableModel) {
        String query = "SELECT DISTINCT s.StoreName, c.CameraName, c.Resolution " +
                       "FROM STORE s " +
                       "JOIN CAMERA c ON s.StoreID = c.StoreID " +
                       "JOIN MANUFACTURER m ON c.ManufacturerID = m.ManufacturerID " +
                       "WHERE m.ManufacturerName = ? ORDER BY s.StoreName";
        
        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(query)) {
            
            pstmt.setString(1, manufacturerName);
            ResultSet rs = pstmt.executeQuery();
            
            tableModel.setRowCount(0);
            while (rs.next()) {
                tableModel.addRow(new Object[]{
                    rs.getString("StoreName"),
                    rs.getString("CameraName"),
                    rs.getString("Resolution")
                });
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

Java Swing GUI Panel

import javax.swing.*;
import javax.swing.table.DefaultTableModel;
import java.awt.*;
import java.awt.event.ActionEvent;

public class LogisticsGUI extends JFrame {
    private JTable resultsTable;
    private DefaultTableModel tableModel;
    private JComboBox querySelector;

    public LogisticsGUI() {
        setTitle("Camera Logistics Operations System");
        setSize(1000, 600);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLocationRelativeTo(null);

        // Setup main panel
        JPanel mainPanel = new JPanel(new BorderLayout());
        
        // Control panel
        JPanel controlPanel = new JPanel(new FlowLayout());
        String[] queries = {
            "Cameras in Store",
            "Count by Manufacturer",
            "Inventory by Store",
            "Parts Suppliers",
            "Supply Chain Detail"
        };
        querySelector = new JComboBox<>(queries);
        controlPanel.add(new JLabel("Select Query:"));
        controlPanel.add(querySelector);
        
        JButton executeBtn = new JButton("Execute");
        executeBtn.addActionListener(e -> executeQuery());
        controlPanel.add(executeBtn);

        // Results table
        tableModel = new DefaultTableModel();
        resultsTable = new JTable(tableModel);
        JScrollPane scrollPane = new JScrollPane(resultsTable);

        mainPanel.add(controlPanel, BorderLayout.NORTH);
        mainPanel.add(scrollPane, BorderLayout.CENTER);

        add(mainPanel);
        setVisible(true);
    }

    private void executeQuery() {
        int queryIndex = querySelector.getSelectedIndex();
        
        switch(queryIndex) {
            case 0: // Cameras in Store
                setupCameraTableColumns();
                LogisticsOperationsSystem.getCamerasInStore(1, tableModel);
                break;
            case 1: // Count by Manufacturer
                setupManufacturerCountColumns();
                LogisticsOperationsSystem.getCameraCountByManufacturer(tableModel);
                break;
            case 2: // Inventory by Store
                setupStoreInventoryColumns();
                LogisticsOperationsSystem.getInventoryByStore(tableModel);
                break;
        }
        resultsTable.repaint();
    }

    private void setupCameraTableColumns() {
        tableModel.setColumnIdentifiers(new String[]{"Camera ID", "Name", "Resolution", "Manufacturer"});
    }

    private void setupManufacturerCountColumns() {
        tableModel.setColumnIdentifiers(new String[]{"Manufacturer", "Camera Count"});
    }

    private void setupStoreInventoryColumns() {
        tableModel.setColumnIdentifiers(new String[]{"Store", "City", "State", "Inventory"});
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(LogisticsGUI::new);
    }
}
